#!/bin/bash

#grab a list of every process running,
#grab list of every new process running,
#give us the diff
#repeat

# Loop by line, not space
IFS=$'\n'

old_process=$(ps -eo command)
while true; do
  new_process=$(ps -eo command)
  deff <(echo "$old_process") <(echo "$new_process") | grep [\<\<]
  sleep 1
  old_process=$new_process
done
